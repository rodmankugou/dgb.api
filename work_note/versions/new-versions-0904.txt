CREATE TABLE `asset_record` (
  `id` bigint NOT NULL AUTO_INCREMENT,
  `chain_tx_id` bigint NOT NULL,
  `tx_hash` varchar(255) DEFAULT NULL,
  `chain_tx_type` tinyint DEFAULT NULL,
  `block_height` bigint DEFAULT NULL,
  `block_time` bigint DEFAULT NULL,
  `type` int DEFAULT NULL COMMENT '11-版本1的叫价扣款 12-版本2的叫价扣款 13-版本3的叫价扣款 21-版本1的叫价退款 22-版本2的叫价退款 23-版本3的叫价退款 24-取消重复拍卖退款 3-分润 4-市场费用 5-项目方分润 6-版税池 7-售卖者收入 100-NFT转让',
  `artwork_id` bigint DEFAULT NULL,
  `token` varchar(255) DEFAULT NULL,
  `token_id` varchar(255) DEFAULT NULL,
  `auction_record_id` bigint DEFAULT NULL,
  `ex_contract_address` varchar(255) DEFAULT NULL,
  `chain_order_id` varchar(255) DEFAULT NULL,
  `bid_record_id` bigint DEFAULT NULL,
  `bid_order_id` varchar(255) DEFAULT NULL,
  `from_address` varchar(255) DEFAULT NULL,
  `payment_address` varchar(255) DEFAULT NULL,
  `receive_address` varchar(255) DEFAULT NULL,
  `amount` decimal(38,18) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `chain_tx_id` (`chain_tx_id`) USING BTREE,
  KEY `tx_hash` (`tx_hash`) USING BTREE,
  KEY `auction_record_id` (`auction_record_id`) USING BTREE,
  KEY `artwork_id` (`artwork_id`),
  KEY `bid_id` (`bid_record_id`),
  KEY `payment_address` (`payment_address`),
  KEY `receive_address` (`receive_address`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
